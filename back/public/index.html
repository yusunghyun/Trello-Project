cd front 후 npm i axios@0.19.0 import * as api from '../api' 로 임폴트

api.auth.login(email,password)
Logout
api.auth.logout()
Join
api.auth.join(email,password)
Get board list
api.board.fetch()

api.board.fetch(id)
Add board
api.board.create(title)
Edit board
api.board.update(id,{title,bgColor})
Delete board
api.board.destroy(id)

api.list.create({title,BoardId,pos})
Edit list

Delete list
api.list.destroy(id)

api.card.create(title,ListId,pos)
